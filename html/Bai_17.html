def muscl_reconstr(Ps):
    P_left = Ps[:-1].copy()
    P_right = Ps[1:].copy()
    rs = (Ps[1:-1] - Ps[:-2])/(Ps[2:] - Ps[1:-1] + 1e-10)
    
    P_left[1:] = Ps[1:-1] + 0.5*minmod(rs[:])*(Ps[2:] - Ps[1:-1])
    P_left[0] = Ps[0] #iu kin bin 
    
    P_right[:-1] = Ps[1:-1] - 0.5*minmod(rs[:])*(Ps[2:] - Ps[1:-1])
    P_right[-1] = Ps[-1] #iu kin bin
    return P_left, P_right

#P=(r, u, p): amax(a, P) = (max(a, r), max(a, u), max(a, p))
def amax(a, P):
    return np.array([max(a, pi) for pi in P])

def amin(a, P):
    return np.array([min(a, pi) for pi in P])

def minmod(rs):
    phi = np.array([amax(0, amin(1, r)) for r in rs])
    return phi

#s  WENO5
def psum(a, P3):
    return a[0]*P3[0] + a[1]*P3[1] + a[2]*P3[2]

#Hm ti cu trc nghim
def reconstr(Ps, beta, a_b, gamma, a_P, Ps_05):
    for i in range(2, len(Ps)-2): #for each cell in cells
        #bc 1. tnh ch th  trn trong tng khun S_j
        b0 = beta(a_b[0], Ps[i-2:i+1])
        b1 = beta(a_b[1], Ps[i-1:i+2])
        b2 = beta(a_b[2], Ps[i  :i+3])
        
        #bc 2. tm trng lng phi tuyn tng khun
        omg_r = omega(gamma, np.array([b0[0], b1[0], b2[0]]))
        omg_u = omega(gamma, np.array([b0[1], b1[1], b2[1]]))
        omg_p = omega(gamma, np.array([b0[2], b1[2], b2[2]]))
    
        #bc 3. tm xp x bc 3
        P0 = psum(a_P[0], Ps[i-2:i+1])
        P1 = psum(a_P[1], Ps[i-1:i+2])
        P2 = psum(a_P[2], Ps[i  :i+3])
        
        #bc 4. tm xp x bc 5
        Ps_05[i, 0] = sum(omg_r*[P0[0], P1[0], P2[0]])
        Ps_05[i, 1] = sum(omg_u*[P0[1], P1[1], P2[1]])
        Ps_05[i, 2] = sum(omg_p*[P0[2], P1[2], P2[2]])
        
        #iu kin bin: u_05[0], u_05[1], u_05[-2], u_05[-1]
    return

def weno5_reconstr(Ps):
    P_left = Ps.copy()
    P_right = Ps.copy()
    reconstr(Ps, beta_Rct, a_bRct, gamma_Rct_p05, a_uRct_p05, P_left)
    reconstr(Ps, beta_Rct, a_bRct, gamma_Rct_m05, a_uRct_m05, P_right)
    
    return P_left[:-1], P_right[1:]
